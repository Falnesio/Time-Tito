lista_de_numeros = range(0, 20, 3) # de 0 a 20 de 3 em 3

#para imprimir cada

for número in lista_de_numeros:
    print(números)

# Imprimir os nomes dentro de 0 a 3 usando range; 
# já tendo lista certinha não precisa dessa complicação
for i in range(4):
